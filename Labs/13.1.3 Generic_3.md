# 13.1.3 Generic


##  13.1.3.1 Generic class : stack for Integer

```cs
namespace GenericExample
{
    class Student
    {
        public string Name { get; set; }
    }
    internal class Program
    {
        public static void multipleTypeTemplate<S, T>(S s, T t)
        {
            Console.WriteLine($"Type of S is {s?.GetType().Name}, value = {s}");
            Console.WriteLine($"Type of T is {t?.GetType().Name}, value = {t}");
        }

        static void Main(string[] args)
        {
            multipleTypeTemplate<int, float>(1,2.5f);
            Console.WriteLine("-------------");
            multipleTypeTemplate<string, bool>("Hello World",true);
            Console.WriteLine("-------------");
            multipleTypeTemplate<string, bool>("Hello World", true);
            Console.WriteLine("-------------");
            multipleTypeTemplate<Student, int>(new Student() { Name = "Albert"}, 15);

        }
    }
}
```
